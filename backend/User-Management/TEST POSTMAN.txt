CREAR USUARIO
curl -X POST http://localhost:5001/users \
     -H "Content-Type: application/json" \
     -d '{
           "name": "John Doe",
           "email": "john.doe@example.com",
           "phone": "1234567890",
           "password": "securepassword",
           "date_of_birth": "1990-01-01",
           "role": "admin",
           "is_active": true
         }'


OBTENER USUARIO
curl -X GET http://localhost:5003/users/1


OBTENER USUARIOS
curl -X GET http://localhost:5002/users


MODIFICAR USUARIOS
curl -X PUT http://localhost:5008/users/1 \
     -H "Content-Type: application/json" \
     -d '{
           "name": "John Doe Updated",
           "email": "john.doe.updated@example.com",
           "phone": "9876543210",
           "password": "newsecurepassword",
           "date_of_birth": "1990-01-01",
           "role": "admin",
           "is_active": false
         }'


ELIMINAR USUARIO 
curl -X DELETE http://localhost:5005/users/1
